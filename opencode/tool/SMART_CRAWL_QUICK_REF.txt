FIND CHROME PROFILE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
macOS:    ls ~/Library/Application\ Support/Google/Chrome/
Windows:  dir %LOCALAPPDATA%\Google\Chrome\User Data\
Linux:    ls ~/.config/google-chrome/

PARAMETERS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
url            (required)  Target URL
use_chrome     (optional)  Use Chrome profile
chrome_profile (optional)  Profile name (default: "Default")
headless       (optional)  Run invisibly (default: true)
depth          (optional)  Crawl depth (default: 0)
css_schema     (optional)  CSS selectors for extraction

IMPORTANT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
✓ Close Chrome before running (profile lock)
✓ Login to sites in Chrome first
✓ Use headless: false for debugging
✓ Chrome Profile > Cookie Files
